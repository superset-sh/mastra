const agent = new Agent({
  id: "dynamic-agent",
  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "fireworks-ai/accounts/fireworks/models/minimax-m2p5"
      : "fireworks-ai/accounts/fireworks/models/deepseek-v3p1";
  }
});
