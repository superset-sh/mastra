const agent = new Agent({

  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "evroc/openai/whisper-large-v3"
      : "evroc/KBLab/kb-whisper-large";
  }
});
